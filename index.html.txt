    // INITIAL DATA
    let balance = parseInt(localStorage.getItem('huangx_balance')) || 0;
    let tapPower = parseInt(localStorage.getItem('huangx_power')) || 1;

    const maxEnergy = 1000;

    // UPDATE TAMPILAN AWAL
    updateDisplay();

    // LOGIKA KLIK (TAP)
    document.getElementById('coin-image').onclick = function() {
        if (energy > 0) {
            balance += tapPower;
            energy -= 1;
            updateDisplay();
            saveData();
        } else {
            alert("Energy habis! Istirahat dulu bro.");
        }
    };

    // RECHARGE ENERGY (DAPET 3 ENERGY TIAP 2 DETIK)
    setInterval(() => {
        if (energy < maxEnergy) {
            energy += 3;
            if (energy > maxEnergy) energy = maxEnergy;
            updateDisplay();
        }
    }, 2000);

    function updateDisplay() {
        document.getElementById('balance').innerText = balance.toLocaleString();
        document.getElementById('tapPower').innerText = tapPower;
        document.getElementById('energy-text').innerText = energy;
        
        // Update Bar Energy
        let percentage = (energy / maxEnergy) * 100;
        document.getElementById('energy-fill').style.width = percentage + "%";
    }

    function saveData() {
        localStorage.setItem('huangx_balance', balance);
        localStorage.setItem('huangx_power', tapPower);
